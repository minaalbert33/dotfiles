     55 # rename multiple files the way nnn does it (open in editable buffer)
     56 cmd batch_rename !{{
     57   contentsfile=$(mktemp)
     58   trap 'rm $contentsfile' INT TERM EXIT
     59   dircontents="$(command ls -1p)"
     60   printf "%s" "$dircontents" | nl -n ln > "$contentsfile";
     61   $EDITOR "$contentsfile"
     62   printf "%s" "$dircontents" | paste -d '\t' "$contentsfile" - \
     63   | while read -r line; do
     64     orig="$(printf "%s" "$line" | cut -f3 | sed 's!/$!!')"
     65     new="$(printf "%s" "$line" | cut -f2 | sed 's!/$!!')"
     66     if [ "$new" != "$orig" ]; then
     67       if [ -e "$new" ]; then
     68         printf "%s already exists, not overwriting.\n" "$new"
     69       else
     70         printf "./%s -> ./%s\n" "$orig" "$new"
     71         command mv -i ./"$orig" ./"$new"
     72       fi
     73     fi
     74   done
     75   rm "$contentsfile"
     76   trap - INT TERM EXIT
     77 }}

    146 # custom paste command using rsync/mv
    147 cmd paste &{{
    148     load=$(cat ~/.local/share/lf/files)
    149     mode=$(echo "$load" | sed -n '1p')
    150     list=$(echo "$load" | sed '1d')
    151     if [ $mode = 'copy' ]; then
    152         rsync -av --ignore-existing --progress $list . \
    153         | stdbuf -i0 -o0 -e0 tr '\r' '\n' \
    154         | while read line; do
    155             lf -remote "send $id echo $line"
    156         done
    157     elif [ $mode = 'move' ]; then
    158         mv -n $list .
    159     fi
    160     lf -remote 'send load'
    161     lf -remote 'send clear'
    162 }}
